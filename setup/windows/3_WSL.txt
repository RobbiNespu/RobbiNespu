Use your text editor to set the value of ZSH_THEME to "powerlevel10k/powerlevel10k" in ~/.zshrc.
vim ~/.zshrc
p10k configure

# copy your SSH key from Windows to WSL,
# Create .ssh directory in WSL if it doesn't exist
mkdir -p ~/.ssh

# Copy the private key
cp /mnt/c/Users/<YourWindowsUsername>/.ssh/id_rsa ~/.ssh/

# Copy the public key
cp /mnt/c/Users/<YourWindowsUsername>/.ssh/id_rsa.pub ~/.ssh/

# Set correct permissions (important for SSH to work)
chmod 700 ~/.ssh
chmod 600 ~/.ssh/id_rsa
chmod 644 ~/.ssh/id_rsa.pub
# Copy SSH config (optional)
If you have an SSH config file:
bashcp /mnt/c/Users/<YourWindowsUsername>/.ssh/config ~/.ssh/
chmod 600 ~/.ssh/config
